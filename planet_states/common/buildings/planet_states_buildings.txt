building_ps_capitol = {
	capital = yes
	can_build = no
	can_demolish = yes
	icon = building_palace

	potential = {
		exists = owner
		owner = { is_country_type = planet_state }
	}

	convert_to = {
		building_hive_major_capital
		building_machine_major_capital
		building_major_capital
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner

		}
		modifier = {

			planet_amenities_add = 25
		}
	}
